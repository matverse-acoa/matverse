"""
MatVerse Unified Closure + Memory Validator
Valida:
- Pipeline canônico
- Separação Bucket/Repo/Ledger
- Arquitetura cognitiva (LTM/STM/Buffer)
- Risco de fragmentação
"""

from dataclasses import dataclass, asdict
from typing import Dict, List
import hashlib


@dataclass
class SystemState:
    has_bucket_layer: bool
    has_repo_layer: bool
    has_ledger_layer: bool
    omega_unified: bool
    mnb_unified: bool
    replay_deterministic: bool
    external_proof: bool
    memory_layers: Dict[str, bool]  # LTM, STM, Buffer


class UnifiedValidator:

    @staticmethod
    def compute_score(state: SystemState) -> float:
        score = 0.0

        # Arquitetura base
        score += 0.15 if state.has_bucket_layer else 0
        score += 0.15 if state.has_repo_layer else 0
        score += 0.15 if state.has_ledger_layer else 0

        # Governança
        score += 0.15 if state.omega_unified else 0
        score += 0.10 if state.mnb_unified else 0

        # Execução
        score += 0.10 if state.replay_deterministic else 0
        score += 0.10 if state.external_proof else 0

        # Memória cognitiva
        memory_score = sum(state.memory_layers.values()) / 3
        score += 0.10 * memory_score

        return round(score, 3)

    @staticmethod
    def evaluate(state: SystemState) -> Dict:
        score = UnifiedValidator.compute_score(state)

        if score >= 0.9:
            status = "INSTITUTIONALLY_CLOSED"
        elif score >= 0.75:
            status = "NEAR_CLOSURE"
        else:
            status = "OPEN_SYSTEM"

        return {
            "score": score,
            "status": status,
            "recommendation": UnifiedValidator.recommend(status)
        }

    @staticmethod
    def recommend(status: str) -> str:
        if status == "INSTITUTIONALLY_CLOSED":
            return "Sistema pronto para publicação e auditoria externa."
        elif status == "NEAR_CLOSURE":
            return "Falta prova pública independente e replay completo."
        else:
            return "Unificar Ω, MNB e pipeline antes de escalar."

🔧 Artifact Consistency Validator
"""
MatVerse Public Artifact Consistency Validator
Detecta inconsistência entre:
- hashes
- versões
- datasets
- registros públicos
"""

from dataclasses import dataclass
from typing import List, Dict
import hashlib


@dataclass
class Artifact:
    id: str
    hash: str
    source: str  # sepolia, zenodo, hf
    version: str


class ConsistencyValidator:

    @staticmethod
    def hash_equivalence(artifacts: List[Artifact]) -> bool:
        hashes = {a.hash for a in artifacts}
        return len(hashes) == 1

    @staticmethod
    def version_alignment(artifacts: List[Artifact]) -> bool:
        versions = {a.version for a in artifacts}
        return len(versions) == 1

    @staticmethod
    def compute_entropy(artifacts: List[Artifact]) -> float:
        unique_hashes = len(set(a.hash for a in artifacts))
        return round(unique_hashes / len(artifacts), 3)

    @staticmethod
    def evaluate(artifacts: List[Artifact]) -> Dict:
        hash_ok = ConsistencyValidator.hash_equivalence(artifacts)
        version_ok = ConsistencyValidator.version_alignment(artifacts)
        entropy = ConsistencyValidator.compute_entropy(artifacts)

        if hash_ok and version_ok:
            status = "CONSISTENT"
        elif entropy < 0.3:
            status = "PARTIALLY_CONSISTENT"
        else:
            status = "FRAGMENTED"

        return {
            "status": status,
            "hash_consistent": hash_ok,
            "version_consistent": version_ok,
            "entropy": entropy,
            "recommendation": ConsistencyValidator.recommend(status)
        }

    @staticmethod
    def recommend(status: str) -> str:
        if status == "CONSISTENT":
            return "Sistema pronto para fechamento institucional."
        elif status == "PARTIALLY_CONSISTENT":
            return "Normalizar versões e consolidar datasets."
        else:
            return "Alto risco: múltiplas verdades públicas conflitantes."

from __future__ import annotations

import json
import os
import subprocess
from dataclasses import dataclass, asdict
from datetime import datetime, timezone
from pathlib import Path
from typing import Optional


@dataclass
class PipelineConfig:
    project_dir: str
    zenodo_token_env: str = "ZENODO_ACCESS_TOKEN"
    hf_token_env: str = "HUGGINGFACE_TOKEN"
    sepolia_key_env: str = "SEPOLIA_PRIVATE_KEY"


@dataclass
class PipelineReceipt:
    canonical_hash: Optional[str]
    zenodo_doi: Optional[str]
    sepolia_tx: Optional[str]
    hf_links: Optional[str]
    status: str
    timestamp_utc: str
    errors: list[str]


class CanonicalRegistryPipeline:
    def __init__(self, cfg: PipelineConfig) -> None:
        self.cfg = cfg
        self.project_dir = Path(cfg.project_dir).resolve()

    def _run(self, cmd: list[str]) -> subprocess.CompletedProcess:
        return subprocess.run(
            cmd,
            cwd=self.project_dir,
            text=True,
            capture_output=True,
            check=False,
        )

    def _require_files(self, names: list[str]) -> None:
        missing = []
        for name in names:
            p = self.project_dir / name
            if not p.exists() or p.stat().st_size == 0:
                missing.append(name)
        if missing:
            raise FileNotFoundError(f"Artefatos ausentes ou vazios: {missing}")

    def _require_env(self) -> None:
        required = [
            self.cfg.zenodo_token_env,
            self.cfg.hf_token_env,
            self.cfg.sepolia_key_env,
        ]
        missing = [k for k in required if not os.environ.get(k)]
        if missing:
            raise EnvironmentError(f"Variáveis ausentes: {missing}")

    def _read_optional(self, name: str) -> Optional[str]:
        p = self.project_dir / name
        return p.read_text(encoding="utf-8").strip() if p.exists() else None

    def execute(self) -> PipelineReceipt:
        errors: list[str] = []

        try:
            self._require_files([
                "symbios-core.pdf",
                "symbios-core-v1.tar.gz",
                "canonical_hash.txt",
                "zenodo_upload.py",
                "hf_upload.py",
                "sepolia_register.py",
            ])
            self._require_env()
        except Exception as e:
            return PipelineReceipt(
                canonical_hash=None,
                zenodo_doi=None,
                sepolia_tx=None,
                hf_links=None,
                status="BLOCK",
                timestamp_utc=datetime.now(timezone.utc).isoformat(),
                errors=[str(e)],
            )

        for script in ["zenodo_upload.py", "hf_upload.py", "sepolia_register.py"]:
            proc = self._run(["python3", script])
            if proc.returncode != 0:
                errors.append(
                    f"{script} falhou | code={proc.returncode} | stderr={proc.stderr.strip()}"
                )

        receipt = PipelineReceipt(
            canonical_hash=self._read_optional("canonical_hash.txt"),
            zenodo_doi=self._read_optional("zenodo_doi.txt"),
            sepolia_tx=self._read_optional("sepolia_tx.txt"),
            hf_links=self._read_optional("hf_links.txt"),
            status="PASS",
            timestamp_utc=datetime.now(timezone.utc).isoformat(),
            errors=errors,
        )

        if not all([
            receipt.canonical_hash,
            receipt.zenodo_doi,
            receipt.sepolia_tx,
            receipt.hf_links,
        ]):
            receipt.status = "PARTIAL"

        if errors:
            receipt.status = "BLOCK"

        out = self.project_dir / "canonical_registry_receipt.json"
        out.write_text(json.dumps(asdict(receipt), indent=2, ensure_ascii=False), encoding="utf-8")
        return receipt


if __name__ == "__main__":
    cfg = PipelineConfig(project_dir="/home/ubuntu/symbios_project")
    pipeline = CanonicalRegistryPipeline(cfg)
    result = pipeline.execute()
    print(json.dumps(asdict(result), indent=2, ensure_ascii=False))

from __future__ import annotations

from dataclasses import dataclass, asdict
from datetime import datetime, timezone
from hashlib import sha256
from pathlib import Path
from typing import Optional, Dict, Any
import json


@dataclass
class RegistryInput:
    artifact_path: str
    psi: float = 1.0
    cvar: float = 0.0


@dataclass
class RegistryOutput:
    artifact_hash: str
    zenodo_doi: Optional[str]
    hf_dataset: Optional[str]
    hf_space: Optional[str]
    sepolia_tx: Optional[str]
    receipt_root: Optional[str]
    status: str
    timestamp_utc: str
    reasons: list[str]


class CanonicalRegistryValidator:
    def __init__(self, policy_path: str = "POLICY.json") -> None:
        self.policy = json.loads(Path(policy_path).read_text(encoding="utf-8"))

    @staticmethod
    def compute_sha256(path: str) -> str:
        h = sha256()
        with open(path, "rb") as f:
            for chunk in iter(lambda: f.read(8192), b""):
                h.update(chunk)
        return h.hexdigest()

    def validate_input(self, payload: RegistryInput) -> tuple[bool, list[str], str]:
        reasons: list[str] = []
        p = Path(payload.artifact_path)

        if not p.exists():
            reasons.append("artifact_not_found")
            return False, reasons, ""

        if self.policy["require_nonempty_artifact"] and p.stat().st_size == 0:
            reasons.append("artifact_empty")
            return False, reasons, ""

        artifact_hash = self.compute_sha256(payload.artifact_path)

        if self.policy["thresholds"]["psi_min"] > payload.psi:
            reasons.append("psi_below_threshold")

        if payload.cvar > self.policy["thresholds"]["cvar_max"]:
            reasons.append("cvar_above_threshold")

        return len(reasons) == 0, reasons, artifact_hash

    @staticmethod
    def compute_receipt_root(receipt: Dict[str, Any]) -> str:
        canonical = json.dumps(receipt, sort_keys=True, ensure_ascii=False).encode("utf-8")
        return sha256(canonical).hexdigest()

    def final_status(
        self,
        valid_input: bool,
        doi: Optional[str],
        tx: Optional[str],
        hf_dataset: Optional[str],
        hf_space: Optional[str],
        receipt_root: Optional[str]
    ) -> tuple[str, list[str]]:
        reasons: list[str] = []

        if not valid_input:
            reasons.append("policy_fail")
            return "BLOCK", reasons

        anchors = [doi, tx, hf_dataset or hf_space]
        anchor_count = sum(1 for x in anchors if x)

        if receipt_root is None:
            reasons.append("missing_receipt_root")
            return "BLOCK", reasons

        if anchor_count >= self.policy["thresholds"]["min_anchor_count"]:
            return "PASS", reasons

        if anchor_count > 0:
            reasons.append("incomplete_anchor_set")
            return "PARTIAL", reasons

        reasons.append("no_anchor_succeeded")
        return "BLOCK", reasons


def build_receipt(
    artifact_hash: str,
    zenodo_doi: Optional[str],
    hf_dataset: Optional[str],
    hf_space: Optional[str],
    sepolia_tx: Optional[str],
    status: str,
    reasons: list[str],
    parent_hash: Optional[str] = None,
) -> dict:
    return {
        "artifact_hash": artifact_hash,
        "zenodo_doi": zenodo_doi,
        "hf_dataset": hf_dataset,
        "hf_space": hf_space,
        "sepolia_tx": sepolia_tx,
        "status": status,
        "reasons": reasons,
        "parent_hash": parent_hash,
        "timestamp_utc": datetime.now(timezone.utc).isoformat()
    }


def persist_json(obj: dict, path: str) -> None:
    Path(path).write_text(json.dumps(obj, ensure_ascii=False, indent=2), encoding="utf-8")

# Rollback Procedure

## Quando usar
Executar se o status final for PARTIAL ou BLOCK.

## Casos
1. Zenodo passou, HF/Sepolia falharam
2. HF passou, Zenodo/Sepolia falharam
3. Sepolia passou, Zenodo/HF falharam
4. Policy fail antes das âncoras

## Ações
- Não declarar publicação canônica completa
- Marcar receipt como PARTIAL ou BLOCK
- Registrar qual âncora passou e qual falhou
- Reexecutar apenas etapas faltantes se o artifact_hash permanecer idêntico
- Se o artefato mudar, reiniciar pipeline desde o hash
- Nunca reutilizar DOI/tx/hash de outra versão

## Regra soberana
Se artifact_hash mudou, o artefato é outro.
