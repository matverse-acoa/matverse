import random
import numpy as np

class AdversaryRL:
    def __init__(self):
        self.q = np.zeros((10, 4))
        self.alpha = 0.1
        self.gamma = 0.9
        self.epsilon = 0.2

    def discretize(self, omega):
        return min(9, int(omega * 10))

    def choose_action(self, omega):
        s = self.discretize(omega)
        if random.random() < self.epsilon:
            return random.randint(0, 3)
        return np.argmax(self.q[s])

    def update(self, omega, action, reward, omega_next):
        s = self.discretize(omega)
        s_next = self.discretize(omega_next)

        best_next = np.max(self.q[s_next])
        self.q[s][action] += self.alpha * (
            reward + self.gamma * best_next - self.q[s][action]
        )
