"""
MatVerse Closure Validator
Verifica se o sistema atingiu estado de fechamento institucional.
"""

from dataclasses import dataclass
from typing import Dict
import hashlib


@dataclass
class SystemState:
    genesis_hash: str
    merkle_root: str
    replay_ok: bool
    omega_consistent: bool
    external_verification: bool


class ClosureValidator:

    @staticmethod
    def verify_hash_integrity(genesis_hash: str, merkle_root: str) -> bool:
        """
        Verifica consistência básica entre genesis e merkle root
        """
        combined = hashlib.sha256(
            (genesis_hash + merkle_root).encode()
        ).hexdigest()
        return len(combined) == 64

    @staticmethod
    def compute_closure_score(state: SystemState) -> float:
        """
        Score de fechamento institucional (0 → 1)
        """
        score = 0.0

        score += 0.25 if state.genesis_hash else 0
        score += 0.25 if state.replay_ok else 0
        score += 0.2 if state.omega_consistent else 0
        score += 0.15 if state.external_verification else 0
        score += 0.15 if ClosureValidator.verify_hash_integrity(
            state.genesis_hash, state.merkle_root
        ) else 0

        return score

    @staticmethod
    def evaluate(state: SystemState) -> Dict:
        score = ClosureValidator.compute_closure_score(state)

        if score >= 0.9:
            status = "CLOSED"
        elif score >= 0.7:
            status = "NEAR_CLOSURE"
        else:
            status = "OPEN"

        return {
            "closure_score": round(score, 3),
            "status": status,
            "recommendation": ClosureValidator.recommend(status)
        }

    @staticmethod
    def recommend(status: str) -> str:
        if status == "CLOSED":
            return "Sistema pronto para publicação institucional."
        elif status == "NEAR_CLOSURE":
            return "Falta evidência pública independente."
        else:
            return "Falta genesis único e replay determinístico completo."


# =========================
# Exemplo de uso
# =========================

if __name__ == "__main__":
    state = SystemState(
        genesis_hash="abc123",
        merkle_root="def456",
        replay_ok=True,
        omega_consistent=True,
        external_verification=False
    )

    result = ClosureValidator.evaluate(state)
    print(result)

4. Python para Sistemas
A peça mais útil agora é um auditor de coerência constitucional do corpus, para medir quatro coisas automaticamente: aderência ao pipeline canônico, presença da tríade, unicidade do Ω e unicidade do schema MNB. Abaixo está um esqueleto funcional, auditável e pronto para produção local.

from __future__ import annotations

from dataclasses import dataclass, asdict
from pathlib import Path
import re
import json
from typing import Dict, List, Tuple


CANONICAL_PIPELINE = ["PRIMORDIAL", "SVCA", "MNB", "MOTOR", "LEDGER"]

OMEGA_PATTERNS = [
    r"Ω\s*=\s*0\.4\s*\*?\s*Ψ\s*\+\s*0\.3\s*\*?\s*\\hat\{?Θ\}?\s*\+\s*0\.2\s*\*?\s*\(1\s*-\s*CVaR\)\s*\+\s*0\.1\s*\*?\s*PoLE",
    r"Omega\s*=\s*0\.4.*0\.3.*0\.2.*0\.1",
]

MNB_PATTERNS = [
    r"m\s*=\s*\(\s*e\s*,\s*Ψ\s*,\s*C\s*,\s*τ\s*,\s*h\s*\)",
    r"m_i\s*=\s*\(\s*e_i\s*,\s*Ψ_i\s*,\s*C_i\s*,\s*τ_i\s*,\s*h_i\s*\)",
]


@dataclass
class FileAudit:
    path: str
    pipeline_score: float
    triad_present: bool
    omega_mentions: int
    canonical_omega_like: int
    mnb_mentions: int
    canonical_mnb_like: int
    warnings: List[str]


@dataclass
class CorpusAudit:
    total_files: int
    mean_pipeline_score: float
    triad_coverage: float
    omega_fragmentation_risk: float
    mnb_fragmentation_risk: float
    files: List[FileAudit]
    verdict: str


def read_text(path: Path) -> str:
    try:
        return path.read_text(encoding="utf-8", errors="ignore")
    except Exception:
        return ""


def score_pipeline(text: str) -> float:
    text_u = text.upper()
    positions = []
    for token in CANONICAL_PIPELINE:
        idx = text_u.find(token)
        positions.append(idx)

    found = sum(1 for p in positions if p >= 0)
    if found == 0:
        return 0.0

    ordered = 0
    last = -1
    for p in positions:
        if p >= 0 and p > last:
            ordered += 1
            last = p

    coverage = found / len(CANONICAL_PIPELINE)
    order_bonus = ordered / len(CANONICAL_PIPELINE)
    return round(0.7 * coverage + 0.3 * order_bonus, 3)


def has_triad(text: str) -> bool:
    text_u = text.upper()
    return all(k in text_u for k in ["BODY-D", "BODY-A", "BODY-X"])


def count_matches(text: str, patterns: List[str]) -> int:
    total = 0
    for pattern in patterns:
        total += len(re.findall(pattern, text, flags=re.IGNORECASE))
    return total


def audit_file(path: Path) -> FileAudit:
    text = read_text(path)

    pipeline_score = score_pipeline(text)
    triad = has_triad(text)

    omega_mentions = len(re.findall(r"[ΩO]mega|Ω-GATE|OMEGA-GATE", text, flags=re.IGNORECASE))
    canonical_omega_like = count_matches(text, OMEGA_PATTERNS)

    mnb_mentions = len(re.findall(r"\bMNB\b|Mem-?Nano-?Bit", text, flags=re.IGNORECASE))
    canonical_mnb_like = count_matches(text, MNB_PATTERNS)

    warnings: List[str] = []
    if pipeline_score < 0.6:
        warnings.append("pipeline canônico fraco ou incompleto")
    if not triad:
        warnings.append("tríade Body-D/A/X ausente")
    if omega_mentions > 0 and canonical_omega_like == 0:
        warnings.append("Ω mencionado sem forma canônica reconhecida")
    if mnb_mentions > 0 and canonical_mnb_like == 0:
        warnings.append("MNB mencionado sem schema mínimo reconhecido")

    return FileAudit(
        path=str(path),
        pipeline_score=pipeline_score,
        triad_present=triad,
        omega_mentions=omega_mentions,
        canonical_omega_like=canonical_omega_like,
        mnb_mentions=mnb_mentions,
        canonical_mnb_like=canonical_mnb_like,
        warnings=warnings,
    )


def audit_corpus(base_dir: str) -> CorpusAudit:
    base = Path(base_dir)
    files = sorted([p for p in base.rglob("*") if p.is_file() and p.suffix.lower() in {".txt", ".md", ".py"}])

    audits = [audit_file(p) for p in files]
    total = len(audits)

    if total == 0:
        return CorpusAudit(
            total_files=0,
            mean_pipeline_score=0.0,
            triad_coverage=0.0,
            omega_fragmentation_risk=1.0,
            mnb_fragmentation_risk=1.0,
            files=[],
            verdict="EMPTY",
        )

    mean_pipeline = round(sum(a.pipeline_score for a in audits) / total, 3)
    triad_coverage = round(sum(1 for a in audits if a.triad_present) / total, 3)

    omega_total_mentions = sum(a.omega_mentions for a in audits)
    omega_total_canonical = sum(a.canonical_omega_like for a in audits)
    omega_fragmentation = 1.0 if omega_total_mentions == 0 else round(
        1.0 - min(1.0, omega_total_canonical / max(1, omega_total_mentions)), 3
    )

    mnb_total_mentions = sum(a.mnb_mentions for a in audits)
    mnb_total_canonical = sum(a.canonical_mnb_like for a in audits)
    mnb_fragmentation = 1.0 if mnb_total_mentions == 0 else round(
        1.0 - min(1.0, mnb_total_canonical / max(1, mnb_total_mentions)), 3
    )

    if mean_pipeline >= 0.8 and triad_coverage >= 0.6 and omega_fragmentation <= 0.4 and mnb_fragmentation <= 0.4:
        verdict = "CANONICAL_STRONG"
    elif mean_pipeline >= 0.6:
        verdict = "CANONICAL_PARTIAL"
    else:
        verdict = "FRAGMENTED"

    return CorpusAudit(
        total_files=total,
        mean_pipeline_score=mean_pipeline,
        triad_coverage=triad_coverage,
        omega_fragmentation_risk=omega_fragmentation,
        mnb_fragmentation_risk=mnb_fragmentation,
        files=audits,
        verdict=verdict,
    )


if __name__ == "__main__":
    result = audit_corpus("/mnt/data")
    print(json.dumps(asdict(result), indent=2, ensure_ascii=False))

## 🔧 engine.py

python id="xengine1"
import uuid
from datetime import datetime

MEMORY = []
RULES = []

def generate(prompt):
    return {"code": f"app built for: {prompt}"}

def validate(system):
    return True

def deploy(system):
    return f"https://app-{uuid.uuid4().hex[:6]}.vercel.app"

def monitor():
    return {
        "latency": 200,
        "errors": 0
    }

def learn(prompt, result, metrics):
    MEMORY.append({
        "prompt": prompt,
        "result": result,
        "metrics": metrics
    })

def extract_rule():
    failures = [m for m in MEMORY if m["metrics"]["errors"] > 0]
    if not failures:
        return None

    return "Fix dependency issues"

def improve(prompt, rule):
    if not rule:
        return prompt
    return f"{prompt}\nFix:\n{rule}"

def run(prompt):
    for attempt in range(3):
        system = generate(prompt)

        if not validate(system):
            continue

        url = deploy(system)
        metrics = monitor()

        learn(prompt, system, metrics)

        if metrics["errors"] == 0:
            return {
                "status": "success",
                "url": url,
                "attempts": attempt + 1
            }

        rule = extract_rule()
        prompt = improve(prompt, rule)

    return {
        "status": "failed",
        "attempts": 3
    }

from __future__ import annotations

from dataclasses import dataclass, asdict, field
from datetime import datetime, timezone
from hashlib import sha256
from pathlib import Path
from typing import Any, Dict, List, Optional
import json
import statistics
import time
import uuid


# ============================================================
# 1. MODELOS CANÔNICOS
# ============================================================

@dataclass
class Policy:
    psi_threshold: float = 0.80
    cvar_limit: float = 0.20
    max_retries: int = 3
    auto_rebuild: bool = True
    learning_enabled: bool = True


@dataclass
class GeneratedSystem:
    prompt: str
    code: str
    version_id: str
    created_at: str


@dataclass
class ValidationResult:
    passed: bool
    psi: float
    reasons: List[str] = field(default_factory=list)


@dataclass
class DeployResult:
    success: bool
    url: Optional[str]
    provider: str
    deploy_id: str
    reasons: List[str] = field(default_factory=list)


@dataclass
class MonitorMetrics:
    latency_ms: float
    errors: int
    success_rate: float
    samples: List[float] = field(default_factory=list)


@dataclass
class LearningRule:
    category: str
    rule: str
    confidence: float


@dataclass
class Receipt:
    receipt_id: str
    parent_receipt_id: Optional[str]
    timestamp_utc: str
    attempt: int
    prompt_hash: str
    version_id: str
    status: str
    psi: float
    latency_ms: float
    errors: int
    success_rate: float
    cvar_proxy: float
    omega: float
    url: Optional[str]
    reasons: List[str]


# ============================================================
# 2. ENGINE
# ============================================================

class MatforgeAutonomousEngine:
    def __init__(self, policy: Policy, storage_dir: str = "./matforge_engine_state") -> None:
        self.policy = policy
        self.storage = Path(storage_dir)
        self.storage.mkdir(parents=True, exist_ok=True)

        self.memory_file = self.storage / "memory.jsonl"
        self.rules_file = self.storage / "rules.json"
        self.receipts_file = self.storage / "receipts.jsonl"

        self.rules: List[LearningRule] = self._load_rules()

    # --------------------------------------------------------
    # Estado / Persistência
    # --------------------------------------------------------

    def _load_rules(self) -> List[LearningRule]:
        if not self.rules_file.exists():
            return []
        try:
            raw = json.loads(self.rules_file.read_text(encoding="utf-8"))
            return [LearningRule(**item) for item in raw]
        except Exception:
            return []

    def _save_rules(self) -> None:
        payload = [asdict(rule) for rule in self.rules]
        self.rules_file.write_text(
            json.dumps(payload, ensure_ascii=False, indent=2),
            encoding="utf-8",
        )

    def _append_jsonl(self, path: Path, obj: Dict[str, Any]) -> None:
        with path.open("a", encoding="utf-8") as f:
            f.write(json.dumps(obj, ensure_ascii=False) + "\n")

    @staticmethod
    def _now_utc() -> str:
        return datetime.now(timezone.utc).isoformat()

    @staticmethod
    def _hash_text(text: str) -> str:
        return sha256(text.encode("utf-8")).hexdigest()

    # --------------------------------------------------------
    # Etapa 1 - Generate
    # --------------------------------------------------------

    def generate(self, prompt: str) -> GeneratedSystem:
        code = (
            "# generated_app.py\n"
            f"# prompt: {prompt}\n\n"
            "def app():\n"
            f"    return 'app built for: {prompt}'\n"
        )
        return GeneratedSystem(
            prompt=prompt,
            code=code,
            version_id=str(uuid.uuid4()),
            created_at=self._now_utc(),
        )

    # --------------------------------------------------------
    # Etapa 2 - Validate
    # --------------------------------------------------------

    def validate(self, system: GeneratedSystem) -> ValidationResult:
        reasons: List[str] = []

        code = system.code
        score = 0.0

        if "def app(" in code:
            score += 0.30
        else:
            reasons.append("função principal ausente")

        if len(code) > 40:
            score += 0.20
        else:
            reasons.append("código trivial demais")

        if "return" in code:
            score += 0.20
        else:
            reasons.append("sem retorno explícito")

        if "generated_app.py" in code:
            score += 0.15
        else:
            reasons.append("estrutura de arquivo ausente")

        if len(system.prompt.strip()) >= 10:
            score += 0.15
        else:
            reasons.append("prompt fraco ou curto demais")

        passed = score >= self.policy.psi_threshold
        return ValidationResult(passed=passed, psi=round(score, 3), reasons=reasons)

    # --------------------------------------------------------
    # Etapa 3 - Deploy
    # --------------------------------------------------------

    def deploy(self, system: GeneratedSystem) -> DeployResult:
        # Adapter local de deploy. Em produção, substituir por provedor real.
        deploy_dir = self.storage / "deployments" / system.version_id
        deploy_dir.mkdir(parents=True, exist_ok=True)

        app_file = deploy_dir / "generated_app.py"
        app_file.write_text(system.code, encoding="utf-8")

        deploy_id = str(uuid.uuid4())
        url = f"file://{deploy_dir.resolve()}"

        return DeployResult(
            success=True,
            url=url,
            provider="local_filesystem_adapter",
            deploy_id=deploy_id,
            reasons=[],
        )

    # --------------------------------------------------------
    # Etapa 4 - Monitor
    # --------------------------------------------------------

    def monitor(self, deploy_result: DeployResult) -> MonitorMetrics:
        # Simulação controlada baseada no histórico, não mock constante.
        base_samples = [160.0, 180.0, 220.0, 210.0, 190.0]

        past_errors = []
        if self.memory_file.exists():
            with self.memory_file.open("r", encoding="utf-8") as f:
                for line in f:
                    try:
                        obj = json.loads(line)
                        metrics = obj.get("metrics", {})
                        past_errors.append(metrics.get("errors", 0))
                    except Exception:
                        continue

        recent_error_rate = (sum(past_errors[-5:]) / max(1, len(past_errors[-5:]))) if past_errors else 0.0
        errors = 1 if recent_error_rate > 0.4 else 0
        latency_ms = statistics.mean(base_samples) + (25.0 if errors else 0.0)

        success_rate = 1.0 if errors == 0 else 0.75

        return MonitorMetrics(
            latency_ms=round(latency_ms, 2),
            errors=errors,
            success_rate=round(success_rate, 3),
            samples=base_samples,
        )

    # --------------------------------------------------------
    # Etapa 5 - Risk / Governance
    # --------------------------------------------------------

    @staticmethod
    def compute_cvar_proxy(metrics: MonitorMetrics) -> float:
        if not metrics.samples:
            return 1.0
        sorted_samples = sorted(metrics.samples)
        tail_size = max(1, int(len(sorted_samples) * 0.2))
        tail = sorted_samples[-tail_size:]
        cvar_proxy = sum(tail) / len(tail) / 1000.0
        return round(min(1.0, cvar_proxy), 3)

    @staticmethod
    def normalize_theta(latency_ms: float, theta0: float = 200.0) -> float:
        return round(1.0 / (1.0 + latency_ms / theta0), 3)

    def compute_omega(self, psi: float, latency_ms: float, cvar_proxy: float, learned: bool) -> float:
        theta_hat = self.normalize_theta(latency_ms)
        pole_like = 1.0 if learned else 0.0
        omega = 0.4 * psi + 0.3 * theta_hat + 0.2 * (1.0 - cvar_proxy) + 0.1 * pole_like
        return round(omega, 3)

    # --------------------------------------------------------
    # Etapa 6 - Learn
    # --------------------------------------------------------

    def learn(self, prompt: str, system: GeneratedSystem, validation: ValidationResult, deploy: DeployResult, metrics: MonitorMetrics) -> Optional[LearningRule]:
        memory_record = {
            "timestamp_utc": self._now_utc(),
            "prompt": prompt,
            "version_id": system.version_id,
            "validation": asdict(validation),
            "deploy": asdict(deploy),
            "metrics": asdict(metrics),
        }
        self._append_jsonl(self.memory_file, memory_record)

        if not self.policy.learning_enabled:
            return None

        if metrics.errors > 0:
            rule = LearningRule(
                category="runtime_failure",
                rule="Reduzir dependências externas, endurecer health-check e validar entrypoint antes do deploy.",
                confidence=0.82,
            )
            self.rules.append(rule)
            self._save_rules()
            return rule

        return None

    # --------------------------------------------------------
    # Etapa 7 - Improve
    # --------------------------------------------------------

    def improve(self, prompt: str, rule: Optional[LearningRule]) -> str:
        if rule is None:
            return prompt

        return (
            f"{prompt}\n\n"
            f"[AUTO-IMPROVEMENT]\n"
            f"Categoria: {rule.category}\n"
            f"Regra: {rule.rule}\n"
            f"Confiança: {rule.confidence}\n"
        )

    # --------------------------------------------------------
    # Etapa 8 - Receipt
    # --------------------------------------------------------

    def emit_receipt(
        self,
        parent_receipt_id: Optional[str],
        attempt: int,
        prompt: str,
        system: GeneratedSystem,
        validation: ValidationResult,
        deploy: DeployResult,
        metrics: MonitorMetrics,
        learned: bool,
        status: str,
        reasons: List[str],
    ) -> Receipt:
        cvar_proxy = self.compute_cvar_proxy(metrics)
        omega = self.compute_omega(validation.psi, metrics.latency_ms, cvar_proxy, learned)

        receipt = Receipt(
            receipt_id=str(uuid.uuid4()),
            parent_receipt_id=parent_receipt_id,
            timestamp_utc=self._now_utc(),
            attempt=attempt,
            prompt_hash=self._hash_text(prompt),
            version_id=system.version_id,
            status=status,
            psi=validation.psi,
            latency_ms=metrics.latency_ms,
            errors=metrics.errors,
            success_rate=metrics.success_rate,
            cvar_proxy=cvar_proxy,
            omega=omega,
            url=deploy.url,
            reasons=reasons,
        )

        self._append_jsonl(self.receipts_file, asdict(receipt))
        return receipt

    # --------------------------------------------------------
    # Etapa 9 - Run
    # --------------------------------------------------------

    def run(self, prompt: str) -> Dict[str, Any]:
        current_prompt = prompt
        parent_receipt_id: Optional[str] = None

        for attempt in range(1, self.policy.max_retries + 1):
            system = self.generate(current_prompt)
            validation = self.validate(system)

            if not validation.passed:
                receipt = self.emit_receipt(
                    parent_receipt_id=parent_receipt_id,
                    attempt=attempt,
                    prompt=current_prompt,
                    system=system,
                    validation=validation,
                    deploy=DeployResult(False, None, "none", "none", ["validação falhou"]),
                    metrics=MonitorMetrics(999.0, 1, 0.0, []),

from __future__ import annotations

from dataclasses import dataclass, asdict
from datetime import datetime, timezone
from hashlib import sha256
from typing import List, Optional, Dict, Any
import heapq
import json
from pathlib import Path


# ============================================================
# 1. MODELOS
# ============================================================

@dataclass
class MNBRecord:
    mnb_id: str
    psi: float
    cvar: float
    theta_ms: float
    pole: int
    timestamp_utc: str
    payload_path: str
    source_layer: str  # bucket | repo | local
    admitted: bool = False


@dataclass
class CognitivePolicy:
    psi_ltm_threshold: float = 0.85
    psi_stm_threshold: float = 0.80
    cvar_limit: float = 0.20
    buffer_size: int = 5


# ============================================================
# 2. MEMÓRIA COGNITIVA
# ============================================================

class CognitiveCore:
    def __init__(self, policy: CognitivePolicy, state_dir: str = "./cognitive_state"):
        self.policy = policy
        self.state_dir = Path(state_dir)
        self.state_dir.mkdir(parents=True, exist_ok=True)

        self.ltm_file = self.state_dir / "ltm.json"
        self.ledger_file = self.state_dir / "ledger.jsonl"

        self.ltm: List[MNBRecord] = self._load_ltm()
        self.stm: List[MNBRecord] = []
        self.buffer: List[MNBRecord] = []

    def _load_ltm(self) -> List[MNBRecord]:
        if not self.ltm_file.exists():
            return []
        raw = json.loads(self.ltm_file.read_text(encoding="utf-8"))
        return [MNBRecord(**item) for item in raw]

    def _save_ltm(self) -> None:
        self.ltm_file.write_text(
            json.dumps([asdict(x) for x in self.ltm], ensure_ascii=False, indent=2),
            encoding="utf-8",
        )

    def _append_ledger(self, obj: Dict[str, Any]) -> None:
        with self.ledger_file.open("a", encoding="utf-8") as f:
            f.write(json.dumps(obj, ensure_ascii=False) + "\n")

    @staticmethod
    def _now() -> str:
        return datetime.now(timezone.utc).isoformat()

    @staticmethod
    def normalize_theta(theta_ms: float, theta0: float = 200.0) -> float:
        return 1.0 / (1.0 + theta_ms / theta0)

    def omega(self, m: MNBRecord) -> float:
        theta_hat = self.normalize_theta(m.theta_ms)
        return round(
            0.4 * m.psi +
            0.3 * theta_hat +
            0.2 * (1.0 - m.cvar) +
            0.1 * float(m.pole),
            3,
        )

    def admit_to_ltm(self, m: MNBRecord) -> bool:
        if m.psi < self.policy.psi_ltm_threshold:
            return False
        if m.cvar > self.policy.cvar_limit:
            return False
        if self.omega(m) < 0.85:
            return False
        return True

    def ingest(self, m: MNBRecord) -> Dict[str, Any]:
        admitted = self.admit_to_ltm(m)
        m.admitted = admitted

        receipt = {
            "timestamp_utc": self._now(),
            "mnb_id": m.mnb_id,
            "psi": m.psi,
            "cvar": m.cvar,
            "theta_ms": m.theta_ms,
            "pole": m.pole,
            "omega": self.omega(m),
            "source_layer": m.source_layer,
            "admitted": admitted,
            "payload_path": m.payload_path,
        }

        self._append_ledger(receipt)

        if admitted:
            self.ltm.append(m)
            self._save_ltm()

        self.rebuild_stm()
        self.rebuild_buffer()

        return receipt

    def rebuild_stm(self) -> None:
        self.stm = [x for x in self.ltm if x.psi >= self.policy.psi_stm_threshold]

    def rebuild_buffer(self) -> None:
        now = datetime.now(timezone.utc)

        def score(x: MNBRecord) -> float:
            age_seconds = max(
                1.0,
                (now - datetime.fromisoformat(x.timestamp_utc)).total_seconds()
            )
            recency = 1.0 / (1.0 + age_seconds)
            utility = self.omega(x)
            return 0.5 * x.psi + 0.2 * recency + 0.3 * utility

        self.buffer = heapq.nlargest(
            self.policy.buffer_size,
            self.stm,
            key=score,
        )

    def get_active_context(self) -> List[Dict[str, Any]]:
        return [asdict(x) for x in self.buffer]

from datetime import datetime
import uuid
import hashlib

class SovereignLoop:

    def __init__(self, memory, sgsi, ledger):
        self.memory = memory
        self.sgsi = sgsi
        self.ledger = ledger

    def hash_event(self, data):
        return hashlib.sha256(str(data).encode()).hexdigest()

    def run(self, input_text):

        # 1. gerar MNB
        mnb = {
            "id": str(uuid.uuid4()),
            "input": input_text,
            "timestamp": datetime.utcnow().isoformat()
        }

        # 2. avaliar
        metrics = self.sgsi.evaluate(mnb)

        decision = self.sgsi.decide(metrics)

        # 3. fail-closed
        if decision == "FAIL_CLOSED":
            return {"status": "blocked"}

        # 4. memória
        self.memory.add(mnb)
        self.memory.build_stm()
        self.memory.build_buffer()

        context = self.memory.get_context()

        # 5. ledger
        record = {
            "mnb": mnb,
            "metrics": metrics,
            "decision": decision,
            "context_size": len(context)
        }

        record["hash"] = self.hash_event(record)

        self.ledger.append(record)

        return {
            "decision": decision,
            "context": context,
            "hash": record["hash"]
        }

import hashlib
import json
from datetime import datetime


class ClosureEngine:

    def __init__(self, ledger):
        self.ledger = ledger

    # =========================
    # 1. MERKLE ROOT
    # =========================
    def compute_merkle_root(self, records):
        hashes = [
            hashlib.sha256(json.dumps(r).encode()).hexdigest()
            for r in records
        ]

        while len(hashes) > 1:
            new_hashes = []

            for i in range(0, len(hashes), 2):
                left = hashes[i]
                right = hashes[i + 1] if i + 1 < len(hashes) else left

                combined = hashlib.sha256(
                    (left + right).encode()
                ).hexdigest()

                new_hashes.append(combined)

            hashes = new_hashes

        return hashes[0] if hashes else None

    # =========================
    # 2. GENESIS
    # =========================
    def create_genesis(self, config):
        genesis_hash = hashlib.sha256(
            json.dumps(config).encode()
        ).hexdigest()

        block = {
            "type": "GENESIS",
            "hash": genesis_hash,
            "timestamp": datetime.utcnow().isoformat()
        }

        self.ledger.append(block)

        root = self.compute_merkle_root(records)

        block = {
            "type": "MERKLE_ROOT",
            "root": root,
            "timestamp": datetime.utcnow().isoformat()
        }

        self.ledger.append(block)

        return root

    # =========================
    def anchor(self, root):
        # aqui você substitui por Sepolia / RPC real
        tx_hash = hashlib.sha256(root.encode()).hexdigest()

        anchor_record = {
            "type": "ANCHOR",
            "root": root,
            "tx_hash": tx_hash,
            "timestamp": datetime.utcnow().isoformat()
        }

        self.ledger.append(anchor_record)

        return tx_hash

    # 5. FULL CLOSURE
    # =========================
    def close(self, records):
        root = self.build_receipt_block(records)
        tx = self.anchor(root)

            "anchor_tx": tx,

import re
from collections import defaultdict

def normalize_title(title):
    return re.sub(r'[^a-z0-9]', '', title.lower())

def detect_duplicates(records):
    seen = defaultdict(list)

    for r in records:
        key = normalize_title(r["title"])
        seen[key].append(r)

    duplicates = {
        k: v for k, v in seen.items() if len(v) > 1
    }

    return duplicates


def detect_missing_metadata(records):
    issues = []

    for r in records:
        if not r.get("description"):
            issues.append((r["title"], "missing description"))

        if not r.get("authors"):
            issues.append((r["title"], "missing authors"))

    return issues


def classify_records(records):
    classification = defaultdict(list)

    for r in records:
        t = r.get("type", "unknown")

        if "dataset" in t.lower():
            classification["dataset"].append(r)
        elif "software" in t.lower():
            classification["software"].append(r)
        elif "paper" in r["title"].lower():
            classification["paper"].append(r)
        else:
            classification["other"].append(r)

    return classification

def extract_zip(zip_path: Path, out_dir: Path) -> Path:
    target = out_dir / zip_path.stem
    target.mkdir(parents=True, exist_ok=True)
    with zipfile.ZipFile(zip_path, "r") as zf:
        zf.extractall(target)
    return target

def build_manifest(root: Path) -> list[dict]:
    manifest = []
    for p in sorted(root.rglob("*")):
        if p.is_file():
            manifest.append({
                "path": str(p.relative_to(root)),
                "size": p.stat().st_size,
                "suffix": p.suffix.lower(),
                "priority": p.name in PRIORITY_NAMES
            })
    return manifest

def select_priority_files(root: Path) -> list[Path]:
    selected = []
    for p in sorted(root.rglob("*")):
        if not p.is_file():
            continue
        if p.name in PRIORITY_NAMES or p.suffix.lower() in {".py", ".md", ".txt", ".sol"}:
            selected.append(p)
    # prioridade primeiro, depois os demais
    selected.sort(key=lambda x: (x.name not in PRIORITY_NAMES, str(x)))
    return selected

def build_context_bundle(root: Path, output_file: Path, max_chars_per_file: int = 50000) -> None:
    files = select_priority_files(root)
    parts = []

    for f in files:
        try:
            text = f.read_text(encoding="utf-8", errors="ignore")
        except Exception:
            continue

        if len(text) > max_chars_per_file:
            text = text[:max_chars_per_file] + "\n\n[TRUNCATED]\n"

        parts.append(f"===== FILE: {f.relative_to(root)} =====\n{text}\n")

    output_file.write_text("\n".join(parts), encoding="utf-8")

def process_zip(zip_path_str: str, workspace: str = "./zip_exports") -> dict:
    zip_path = Path(zip_path_str)
    out_dir = Path(workspace)
    out_dir.mkdir(parents=True, exist_ok=True)

    extracted = extract_zip(zip_path, out_dir)
    manifest = build_manifest(extracted)

    build_context_bundle(extracted, bundle_path)

    return {

        "extracted_to": str(extracted),
        "manifest": str(manifest_path),
        "bundle": str(bundle_path),
        "file_count": len(manifest)
    }

def compute_sha256(path: str) -> str:
    h = sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return h.hexdigest()


def validate_before_publish(path: str) -> None:
    p = Path(path)
    if not p.exists():
        raise FileNotFoundError(f"Artefato não encontrado: {path}")
    if p.stat().st_size == 0:
        raise ValueError("Artefato vazio. Publicação bloqueada.")


def build_receipt(
    artifact_path: str,
    sepolia_tx: Optional[str],
    zenodo_doi: Optional[str],
    hf_dataset: Optional[str],
    hf_space: Optional[str],
) -> RegistryReceipt:
    artifact_hash = compute_sha256(artifact_path)
    status = "PASS" if all([sepolia_tx, zenodo_doi, hf_dataset, hf_space]) else "PARTIAL"

    return RegistryReceipt(
        artifact_path=artifact_path,
        artifact_hash=artifact_hash,
        sepolia_tx=sepolia_tx,
        zenodo_doi=zenodo_doi,
        hf_dataset=hf_dataset,
        hf_space=hf_space,
        timestamp_utc=datetime.now(timezone.utc).isoformat(),
        status=status,
    )

"""
MatVerse Unified Closure + Memory Validator
Valida:
- Pipeline canônico
- Separação Bucket/Repo/Ledger
- Arquitetura cognitiva (LTM/STM/Buffer)
- Risco de fragmentação
"""

from dataclasses import dataclass, asdict
from typing import Dict, List
import hashlib


@dataclass
class SystemState:
    has_bucket_layer: bool
    has_repo_layer: bool
    has_ledger_layer: bool
    omega_unified: bool
    mnb_unified: bool
    replay_deterministic: bool
    external_proof: bool
    memory_layers: Dict[str, bool]  # LTM, STM, Buffer


class UnifiedValidator:

    @staticmethod
    def compute_score(state: SystemState) -> float:
        score = 0.0

        # Arquitetura base
        score += 0.15 if state.has_bucket_layer else 0
        score += 0.15 if state.has_repo_layer else 0
        score += 0.15 if state.has_ledger_layer else 0

        # Governança
        score += 0.15 if state.omega_unified else 0
        score += 0.10 if state.mnb_unified else 0

        # Execução
        score += 0.10 if state.replay_deterministic else 0
        score += 0.10 if state.external_proof else 0

        # Memória cognitiva
        memory_score = sum(state.memory_layers.values()) / 3
        score += 0.10 * memory_score

        return round(score, 3)

    @staticmethod
    def evaluate(state: SystemState) -> Dict:
        score = UnifiedValidator.compute_score(state)

        if score >= 0.9:
            status = "INSTITUTIONALLY_CLOSED"
        elif score >= 0.75:
            status = "NEAR_CLOSURE"
        else:
            status = "OPEN_SYSTEM"

        return {
            "score": score,
            "status": status,
            "recommendation": UnifiedValidator.recommend(status)
        }

    @staticmethod
    def recommend(status: str) -> str:
        if status == "INSTITUTIONALLY_CLOSED":
            return "Sistema pronto para publicação e auditoria externa."
        elif status == "NEAR_CLOSURE":
            return "Falta prova pública independente e replay completo."
        else:
            return "Unificar Ω, MNB e pipeline antes de escalar."

"""
MatVerse Public Artifact Consistency Validator
Detecta inconsistência entre:
- hashes
- versões
- datasets
- registros públicos
"""

from dataclasses import dataclass
from typing import List, Dict
import hashlib


@dataclass
class Artifact:
    id: str
    hash: str
    source: str  # sepolia, zenodo, hf
    version: str


class ConsistencyValidator:

    @staticmethod
    def hash_equivalence(artifacts: List[Artifact]) -> bool:
        hashes = {a.hash for a in artifacts}
        return len(hashes) == 1

    @staticmethod
    def version_alignment(artifacts: List[Artifact]) -> bool:
        versions = {a.version for a in artifacts}
        return len(versions) == 1

    @staticmethod
    def compute_entropy(artifacts: List[Artifact]) -> float:
        unique_hashes = len(set(a.hash for a in artifacts))
        return round(unique_hashes / len(artifacts), 3)

    @staticmethod
    def evaluate(artifacts: List[Artifact]) -> Dict:
        hash_ok = ConsistencyValidator.hash_equivalence(artifacts)
        version_ok = ConsistencyValidator.version_alignment(artifacts)
        entropy = ConsistencyValidator.compute_entropy(artifacts)

        if hash_ok and version_ok:
            status = "CONSISTENT"
        elif entropy < 0.3:
            status = "PARTIALLY_CONSISTENT"
        else:
            status = "FRAGMENTED"

        return {
            "status": status,
            "hash_consistent": hash_ok,
            "version_consistent": version_ok,
            "entropy": entropy,
            "recommendation": ConsistencyValidator.recommend(status)
        }

    @staticmethod
    def recommend(status: str) -> str:
        if status == "CONSISTENT":
            return "Sistema pronto para fechamento institucional."
        elif status == "PARTIALLY_CONSISTENT":
            return "Normalizar versões e consolidar datasets."
        else:
            return "Alto risco: múltiplas verdades públicas conflitantes."

4. Python para Sistemas
Abaixo está o executor endurecido do skillpipeline, pronto para substituir a execução solta dos scripts. Ele aplica validação, chama cada etapa, coleta saídas e falha fechado.

from __future__ import annotations

import json
import os
import subprocess
from dataclasses import dataclass, asdict
from datetime import datetime, timezone
from pathlib import Path
from typing import Optional


@dataclass
class PipelineConfig:
    project_dir: str
    zenodo_token_env: str = "ZENODO_ACCESS_TOKEN"
    hf_token_env: str = "HUGGINGFACE_TOKEN"
    sepolia_key_env: str = "SEPOLIA_PRIVATE_KEY"


@dataclass
class PipelineReceipt:
    canonical_hash: Optional[str]
    zenodo_doi: Optional[str]
    sepolia_tx: Optional[str]
    hf_links: Optional[str]
    status: str
    timestamp_utc: str
    errors: list[str]


class CanonicalRegistryPipeline:
    def __init__(self, cfg: PipelineConfig) -> None:
        self.cfg = cfg
        self.project_dir = Path(cfg.project_dir).resolve()

    def _run(self, cmd: list[str]) -> subprocess.CompletedProcess:
        return subprocess.run(
            cmd,
            cwd=self.project_dir,
            text=True,
            capture_output=True,
            check=False,
        )

    def _require_files(self, names: list[str]) -> None:
        missing = []
        for name in names:
            p = self.project_dir / name
            if not p.exists() or p.stat().st_size == 0:
                missing.append(name)
        if missing:
            raise FileNotFoundError(f"Artefatos ausentes ou vazios: {missing}")

    def _require_env(self) -> None:
        required = [
            self.cfg.zenodo_token_env,
            self.cfg.hf_token_env,
            self.cfg.sepolia_key_env,
        ]
        missing = [k for k in required if not os.environ.get(k)]
        if missing:
            raise EnvironmentError(f"Variáveis ausentes: {missing}")

    def _read_optional(self, name: str) -> Optional[str]:
        p = self.project_dir / name
        return p.read_text(encoding="utf-8").strip() if p.exists() else None

    def execute(self) -> PipelineReceipt:
        errors: list[str] = []

        try:
            self._require_files([
                "symbios-core.pdf",
                "symbios-core-v1.tar.gz",
                "canonical_hash.txt",
                "zenodo_upload.py",
                "hf_upload.py",
                "sepolia_register.py",
            ])
            self._require_env()
        except Exception as e:
            return PipelineReceipt(
                canonical_hash=None,
                zenodo_doi=None,
                sepolia_tx=None,
                hf_links=None,
                status="BLOCK",
                timestamp_utc=datetime.now(timezone.utc).isoformat(),
                errors=[str(e)],
            )

        for script in ["zenodo_upload.py", "hf_upload.py", "sepolia_register.py"]:
            proc = self._run(["python3", script])
            if proc.returncode != 0:
                errors.append(
                    f"{script} falhou | code={proc.returncode} | stderr={proc.stderr.strip()}"
                )

        receipt = PipelineReceipt(
            canonical_hash=self._read_optional("canonical_hash.txt"),
            zenodo_doi=self._read_optional("zenodo_doi.txt"),
            sepolia_tx=self._read_optional("sepolia_tx.txt"),
            hf_links=self._read_optional("hf_links.txt"),
            status="PASS",
            timestamp_utc=datetime.now(timezone.utc).isoformat(),
            errors=errors,
        )

        if not all([
            receipt.canonical_hash,
            receipt.zenodo_doi,
            receipt.sepolia_tx,
            receipt.hf_links,
        ]):
            receipt.status = "PARTIAL"

        if errors:
            receipt.status = "BLOCK"

        out = self.project_dir / "canonical_registry_receipt.json"
        out.write_text(json.dumps(asdict(receipt), indent=2, ensure_ascii=False), encoding="utf-8")
        return receipt


if __name__ == "__main__":
    cfg = PipelineConfig(project_dir="/home/ubuntu/symbios_project")
    pipeline = CanonicalRegistryPipeline(cfg)
    result = pipeline.execute()
    print(json.dumps(asdict(result), indent=2, ensure_ascii=False))
